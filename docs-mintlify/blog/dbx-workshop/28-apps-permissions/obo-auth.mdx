## 💻 샘플 코드

```python
# OBO (On-Behalf-Of-User) 인증으로 사용자 권한으로 쿼리
from databricks.sdk import WorkspaceClient
from fastapi import Request

def get_obo_client(req: Request):
    user_token = req.headers.get('X-Forwarded-Access-Token')
    return WorkspaceClient(host=os.environ['DATABRICKS_HOST'], token=user_token)

# App SP 인증 (default)
w = WorkspaceClient()  # APP_SP token 자동 사용
```
