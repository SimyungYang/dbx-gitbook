## 💻 샘플 코드

```python
import dlt
from pyspark.sql.functions import col

@dlt.table(comment='Bronze raw events')
def bronze_events():
    return spark.readStream.format('cloudFiles') \
        .option('cloudFiles.format', 'json') \
        .load('/Volumes/workshop/raw/')

@dlt.table(comment='Silver — cleaned')
@dlt.expect_or_drop('valid_id', 'event_id IS NOT NULL')
def silver_events():
    return dlt.read_stream('bronze_events').filter(col('event_type').isNotNull())
```
