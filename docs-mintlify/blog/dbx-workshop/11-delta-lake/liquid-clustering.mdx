```sql
-- Liquid Clustering
-- Delta Lake 핵심 명령 예시:

-- 1) 생성
CREATE TABLE workshop_<name>.default.events (
  event_id BIGINT, event_type STRING, ts TIMESTAMP
) USING DELTA;

-- 2) 변경
UPDATE workshop_<name>.default.events SET event_type = 'click' WHERE event_id = 1;

-- 3) 이력 확인
DESCRIBE HISTORY workshop_<name>.default.events;

-- 4) Time Travel
SELECT * FROM workshop_<name>.default.events VERSION AS OF 0;
```
