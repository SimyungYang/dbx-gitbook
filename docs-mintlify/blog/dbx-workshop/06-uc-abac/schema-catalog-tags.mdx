## 💻 샘플 코드

```sql
-- Tag 정의 + Column에 부여
ALTER TABLE workshop_<name>.default.customers
  ALTER COLUMN ssn SET TAGS ('sensitivity' = 'pii');

-- ABAC Policy (예: PII Column 자동 마스킹)
CREATE POLICY mask_pii
ON CATALOG workshop_<name>
TO ALL
WHEN MATCH TAG sensitivity = 'pii'
APPLY MASK (col) -> CASE
  WHEN is_member('pii-allowed') THEN col
  ELSE '***'
END;
```
